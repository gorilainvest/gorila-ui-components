import { Component, Inject, ViewEncapsulation } from '@angular/core';
import { MAT_DIALOG_DATA } from '@angular/material/dialog';

import { PopupConfirmModel } from './popup-confirm.model';

/**
 * A popup confirmation component with customizable content and actions.
 */
@Component({
  selector: 'mag-popup-confirm',
  templateUrl: './popup-confirm.component.html',
  styleUrls: ['./popup-confirm.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class PopupConfirmComponent {
  /**
   * Empty constructor that receives data from injector.
   *
   * @param data the model for the popup
   */
  constructor(@Inject(MAT_DIALOG_DATA) public data: PopupConfirmModel) {}
}


<ng-container *ngIf="data">

  <h2 mat-dialog-title>{{data.title}}</h2>
  <div *ngFor="let content of data.content" [ngClass]="content.class ? content.class : ''">
    <div *ngIf="content.separator" class='separator-container'>
      <div class='popup-separator'></div>
      <div class='separator-title'> {{content.separator}} </div>
      <div class='popup-separator'></div>
    </div>
    <mat-dialog-content>{{content.text}}</mat-dialog-content>
    <mat-dialog-actions>
      <ng-container *ngFor="let btn of content.actions">
        <button mat-button [mat-dialog-close]="btn.value">
          {{btn.text}}
        </button>
      </ng-container>
    </mat-dialog-actions>
  </div>

</ng-container>
