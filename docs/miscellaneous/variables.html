libs/ui-toolkit/scroll-container/src/lib/scroll-manager.ts

onWindowScroll
Default value : ( event: CdkScrollable, edge: Edge = 'top', previousScroll = 0, minDif = 10, departureOffset = 100, returnOffset = 10 ): ScrollInfo | null => { if (!event) { return null; } const scrollOffset = event.measureScrollOffset(edge) || 0; if (Math.abs(scrollOffset - previousScroll) < minDif) { return null; } if (scrollOffset > departureOffset) { return { scrolled: true, offset: scrollOffset }; } else if (scrollOffset < returnOffset) { return { scrolled: false, offset: scrollOffset }; } return null; }
setupScrollSubscription
Default value : <C extends Scrollable>( activate: boolean, componentRef: C, scrollDispatcher: ScrollDispatcher, windowScrollFn: Function ) => { if (componentRef.scrolledSubs) { componentRef.scrolledSubs.unsubscribe(); } if (!activate) { return; } componentRef.scrolledSubs = scrollDispatcher .scrolled() .pipe(untilDestroyed(componentRef)) .subscribe(windowScrollFn.bind(componentRef)); }

Setups a new scroll subscription, also unsubscribe from previous subscriptions if needed.
